% Script_04_Robustez_MonteCarlo.m: Análise de Robustez pelo Método de Monte
Carlo
clc; clear; close all;

% Carregar configuração
config = load_config();

% ======= CARREGAR RESULTADOS ANTERIORES =======
fprintf('Carregando resultados anteriores...\n');

% Temperatura
if exist(config.FILES.temperatura, 'file')
    temp_load = load(config.FILES.temperatura);
    if isfield(temp_load, 'resultados_temperatura')
        r_data = temp_load.resultados_temperatura.r_data;
        Ti_original = temp_load.resultados_temperatura.Ti_reconstructed;
        Ti_err = temp_load.resultados_temperatura.Ti_data_err;
        best_p_original = temp_load.resultados_temperatura.best_p;
        fprintf('✓ Dados de temperatura carregados\n');
    else
        error('Estrutura de dados de temperatura inválida');
    end
else
    error('Arquivo %s não encontrado. Execute o Script 2 primeiro.', config.FILES.temperatura);
end

% Velocidade
if exist(config.FILES.velocidade, 'file')
    vel_load = load(config.FILES.velocidade);
    if isfield(vel_load, 'resultados_velocidade')
        r_Exp = vel_load.resultados_velocidade.r_Exp;
        Vphi_Exp = vel_load.resultados_velocidade.Vphi_Exp;
        Vphi_Exp_err = vel_load.resultados_velocidade.Vphi_Exp_err;
        fprintf('✓ Dados de velocidade carregados\n');
    else
        error('Estrutura de dados de velocidade inválida');
    end
else
    error('Arquivo %s não encontrado. Execute o Script 1 primeiro.', config.FILES.velocidade);
end

% Carregar resultados do modelo de Helander original
if exist('resultados_helander.mat', 'file')
    hel_load = load('resultados_helander.mat');
    if isfield(hel_load, 'resultados_helander')
        % Carregar também o polinômio de referência e seu IC
        poly_val = hel_load.resultados_helander.poly_val;
        r_poly = linspace(0, config.TCABR.a, 1000)';
        RMSE_best = hel_load.resultados_helander.RMSE_best;
        best_degree = hel_load.resultados_helander.best_degree;
        R2_best_poly = hel_load.resultados_helander.R2_best;
        chi2_red_best_poly = hel_load.resultados_helander.chi2_red_best;
        
        % Calcular IC do polinômio
        upper_V_poly = poly_val + 1.96 * RMSE_best;
        lower_V_poly = poly_val - 1.96 * RMSE_best;
        
        fprintf('✓ Dados do modelo de Helander original carregados\n');
    end
end

% ======= PARÂMETROS DO MONTE CARLO =======
n_iterations = 10000; % 10^4 iterações
a = config.TCABR.a;
R = config.TCABR.R;
epsilon = a/R;
mu0 = config.TCABR.mu0;
Ip = config.TCABR.Ip;

% Função canônica para ajuste de temperatura
Ti_function = @(params, r) (params(1) - params(2)) .* (1 - (r/a).^2).^params(3) + params(2);

% Arrays para armazenar resultados
all_params = zeros(n_iterations, 3); % Tmax, Tmin, n
all_R2 = zeros(n_iterations, 1);
all_RMSE = zeros(n_iterations, 1);
all_residuals = zeros(n_iterations, 1);
all_velocities = zeros(n_iterations, length(r_Exp));
best_residual = Inf;
best_iteration = 0;
best_params = zeros(1, 3);
best_V_helander = [];

% Opções para o ajuste não-linear
options = optimset('Display', 'off', 'MaxIter', 1000, 'TolFun', 1e-8, 'TolX', 1e-8);

% ======= LOOP DE MONTE CARLO =======
fprintf('\n%s\n', repmat('=', 1, 70));
fprintf('ANÁLISE DE ROBUSTEZ - MÉTODO DE MONTE CARLO\n');
fprintf('Iterações: %d\n', n_iterations);
fprintf('%s\n', repmat('=', 1, 70));

progress_bar = waitbar(0, 'Executando simulações de Monte Carlo...');

for iter = 1:n_iterations
    % Atualizar barra de progresso a cada 100 iterações
    if mod(iter, 100) == 0
        waitbar(iter/n_iterations, progress_bar, ...
            sprintf('Progresso: %d/%d iterações', iter, n_iterations));
    end
    
    % 1. Perturbar dados de temperatura dentro das barras de erro
    Ti_perturbed = Ti_original + Ti_err .* randn(size(Ti_original));
    
    % Garantir que as temperaturas sejam positivas
    Ti_perturbed = max(Ti_perturbed, 1); % Mínimo de 1 eV
    
    % 2. Ajustar os parâmetros aos dados perturbados
    initial_guess = [200 + 100*rand(), 20 + 20*rand(), 2 + 3*rand()];
    try
        [params, ~, res] = lsqcurvefit(Ti_function, initial_guess, ...
            r_data, Ti_perturbed, [0, 0, 0.1], [1000, 1000, 10], options);
        
        % 3. Calcular gradiente de temperatura
        Tmax = params(1);
        Tmin = params(2);
        n_val = params(3);
        
        % Discretização fina para cálculo do gradiente
        r_fine = linspace(0.006, a, 500)'; % Começa em 0.006 m
        
        % Gradiente de temperatura (analítico)
        dTidr = n_val * (Tmax - Tmin) * (1 - (r_fine/a).^2).^(n_val-1) .* (-2*r_fine/a^2);
        
        % 4. Calcular velocidade de Helander
        Bpol = -(mu0 * (Ip/(2*pi)) * (r_fine/a^2) .* (2 - (r_fine/a).^2) + 0.001);
        V_helander_fine = (2*epsilon ./ Bpol) .* dTidr;
        
        % Interpolar nos pontos experimentais
        V_helander_interp = interp1(r_fine, V_helander_fine, r_Exp, 'pchip', 'extrap');
        
        % 5. Calcular resíduo em relação aos dados experimentais
        residual = sum(((Vphi_Exp - V_helander_interp) ./ Vphi_Exp_err).^2);
        
        % Armazenar resultados
        all_params(iter, :) = params;
        all_residuals(iter) = residual;
        all_velocities(iter, :) = V_helander_interp;
        
        % Calcular métricas para temperatura
        Ti_fit = Ti_function(params, r_data);
        SSres = sum((Ti_perturbed - Ti_fit).^2);
        SStot = sum((Ti_perturbed - mean(Ti_perturbed)).^2);
        all_R2(iter) = 1 - SSres/SStot;
        all_RMSE(iter) = sqrt(SSres / length(r_data));
        
        % Verificar se é o melhor ajuste
        if residual < best_residual
            best_residual = residual;
            best_iteration = iter;
            best_params = params;
            best_V_helander = V_helander_interp;
            
            % Armazenar também o perfil completo para plotagem
            best_V_full = V_helander_fine;
            best_r_fine = r_fine;
        end
        
    catch
        % Se o ajuste falhar, usar valores padrão
        all_params(iter, :) = initial_guess;
        all_residuals(iter) = Inf;
        all_R2(iter) = 0;
        all_RMSE(iter) = Inf;
    end
end

close(progress_bar);

% ======= ANÁLISE ESTATÍSTICA DOS RESULTADOS =======
% Filtrar iterações com sucesso (resíduo finito)
valid_indices = isfinite(all_residuals) & (all_R2 > 0);
valid_params = all_params(valid_indices, :);
valid_residuals = all_residuals(valid_indices);
valid_R2 = all_R2(valid_indices);
valid_RMSE = all_RMSE(valid_indices);

% Calcular estatísticas
mean_params = mean(valid_params, 1);
std_params = std(valid_params, 0, 1);
median_params = median(valid_params, 1);

% Calcular intervalos de confiança empíricos (5% e 95%)
percentile_5 = prctile(valid_params, 5, 1);
percentile_95 = prctile(valid_params, 95, 1);

% ======= CALCULAR VELOCIDADE DE HELANDER ORIGINAL =======
% Calcular gradiente de temperatura original
Tmax_orig = best_p_original(1);
Tmin_orig = best_p_original(2);
n_orig = best_p_original(3);

r_fine_orig = linspace(0.006, a, 500)'; % Começa em 0.006 m
dTidr_orig = n_orig * (Tmax_orig - Tmin_orig) * (1 - (r_fine_orig/a).^2).^(n_orig-1) .* (-2*r_fine_orig/a^2);
Bpol_orig = -(mu0 * (Ip/(2*pi)) * (r_fine_orig/a^2) .* (2 - (r_fine_orig/a).^2) + 0.001);
V_helander_orig = (2*epsilon ./ Bpol_orig) .* dTidr_orig;

% Interpolar nos pontos experimentais
V_helander_orig_interp = interp1(r_fine_orig, V_helander_orig, r_Exp, 'pchip', 'extrap');

% ======= CALCULAR MÉTRICAS PARA A LEGENDA =======
% Métricas para o Helander original (antes da otimização)
SSres_orig = sum((Vphi_Exp - V_helander_orig_interp).^2);
SStot_orig = sum((Vphi_Exp - mean(Vphi_Exp)).^2);
R2_orig = 1 - SSres_orig / SStot_orig;
RMSE_orig = sqrt(SSres_orig / length(r_Exp));
chi2_orig = sum(((Vphi_Exp - V_helander_orig_interp) ./ Vphi_Exp_err).^2);
chi2_red_orig = chi2_orig / (length(r_Exp) - 3);

% Métricas para o melhor Helander (após otimização)
SSres_best = sum((Vphi_Exp - best_V_helander).^2);
SStot_best = sum((Vphi_Exp - mean(Vphi_Exp)).^2);
R2_best = 1 - SSres_best / SStot_best;
RMSE_best = sqrt(SSres_best / length(r_Exp));
chi2_red_best = best_residual / (length(r_Exp) - 3);

% ======= CALCULAR ADERÊNCIA AO IC POLINOMIAL =======
if exist('upper_V_poly', 'var') && exist('lower_V_poly', 'var')
    % Interpolar o IC nos pontos experimentais
    upper_interp = interp1(r_poly, upper_V_poly, r_Exp, 'pchip', 'extrap');
    lower_interp = interp1(r_poly, lower_V_poly, r_Exp, 'pchip', 'extrap');
    
    % Calcular aderência para Helander original
    inside_IC_orig = sum(V_helander_orig_interp >= lower_interp & V_helander_orig_interp <= upper_interp);
    prop_inside_IC_orig = 100 * inside_IC_orig / length(r_Exp);
    
    % Calcular aderência para melhor Helander
    inside_IC_best = sum(best_V_helander >= lower_interp & best_V_helander <= upper_interp);
    prop_inside_IC_best = 100 * inside_IC_best / length(r_Exp);
end

% ======= IMPRESSÃO DOS RESULTADOS =======
fprintf('\nRESULTADOS DA ANÁLISE DE MONTE CARLO:\n');
fprintf('%s\n', repmat('-', 1, 70));
fprintf('Iterações válidas: %d/%d (%.1f%%)\n', ...
    sum(valid_indices), n_iterations, 100*sum(valid_indices)/n_iterations);
fprintf('Melhor iteração: %d (resíduo mínimo: %.4f)\n', best_iteration, best_residual);
fprintf('%s\n', repmat('-', 1, 70));

fprintf('\nComparação de Métricas:\n');
fprintf('%s\n', repmat('-', 1, 70));
fprintf('%-25s %10s %10s %10s %10s\n', 'Modelo', 'R²', 'χ²_red', 'RMSE (m/s)', 'Aderência IC');
fprintf('%s\n', repmat('-', 1, 70));
fprintf('%-25s %10.4f %10.4f %10.0f %10.1f%%\n', 'Helander Original', R2_orig, chi2_red_orig, RMSE_orig, prop_inside_IC_orig);
fprintf('%-25s %10.4f %10.4f %10.0f %10.1f%%\n', 'Helander Otimizado', R2_best, chi2_red_best, RMSE_best, prop_inside_IC_best);
if exist('R2_best_poly', 'var')
    fprintf('%-25s %10.4f %10.4f %10.0f %10s\n', 'Polinômio Referência', R2_best_poly, chi2_red_best_poly, RMSE_best, '-');
end
fprintf('%s\n', repmat('-', 1, 70));

fprintf('\nParâmetros do melhor ajuste:\n');
fprintf('%s\n', repmat('-', 1, 50));
fprintf('%-10s %12s %12s\n', 'Parâmetro', 'Valor', 'Unidade');
fprintf('%s\n', repmat('-', 1, 50));
fprintf('%-10s %12.2f %12s\n', 'T_max', best_params(1), 'eV');
fprintf('%-10s %12.2f %12s\n', 'T_min', best_params(2), 'eV');
fprintf('%-10s %12.3f %12s\n', 'n', best_params(3), '-');
fprintf('%s\n', repmat('-', 1, 50));

% ======= FIGURA 1: ANÁLISE DE ROBUSTEZ - MELHORADA =======
figure('Position', [50, 50, 1400, 900], 'Name', 'Análise de Robustez - Método de Monte Carlo', 'NumberTitle', 'off');

% Subplot 1: Distribuição de T_max
subplot(2, 3, 1);
histogram(valid_params(:, 1), 30, 'FaceColor', [0.2, 0.4, 0.8], 'EdgeColor', 'k', 'FaceAlpha',
0.8);
xlabel('T_{max} [eV]', 'FontSize', 11, 'FontWeight', 'bold');
ylabel('Frequência', 'FontSize', 11, 'FontWeight', 'bold');

hold on;

box on;
text(0.02, 0.98, 'a)', 'Units', 'normalized', 'FontSize', 12, 'FontWeight', 'bold');

% Subplot 2: Distribuição de T_min
subplot(2, 3, 2);
histogram(valid_params(:, 2), 30, 'FaceColor', [0.2, 0.8, 0.4], 'EdgeColor', 'k', 'FaceAlpha',
0.8);
xlabel('T_{min} [eV]', 'FontSize', 11, 'FontWeight', 'bold');
ylabel('Frequência', 'FontSize', 11, 'FontWeight', 'bold');
title('Distribuição de T_{min}', 'FontSize', 12, 'FontWeight', 'bold');
grid on;
hold on;
line([best_params(2), best_params(2)], ylim, 'Color', 'r', 'LineWidth', 2.5, 'LineStyle', '--');
legend({'Distribuição', 'Melhor valor'}, 'FontSize', 10, 'Location', 'northwest');
box on;
text(0.02, 0.98, 'b)', 'Units', 'normalized', 'FontSize', 12, 'FontWeight', 'bold');

% Subplot 3: Distribuição do expoente n
subplot(2, 3, 3);
histogram(valid_params(:, 3), 30, 'FaceColor', [0.8, 0.2, 0.6], 'EdgeColor', 'k', 'FaceAlpha',
0.8);
xlabel('Expoente n', 'FontSize', 11, 'FontWeight', 'bold');
ylabel('Frequência', 'FontSize', 11, 'FontWeight', 'bold');
title('Distribuição do expoente n', 'FontSize', 12, 'FontWeight', 'bold');
grid on;
hold on;
line([best_params(3), best_params(3)], ylim, 'Color', 'r', 'LineWidth', 2.5, 'LineStyle', '--');
legend({'Distribuição', 'Melhor valor'}, 'FontSize', 10, 'Location', 'northwest');
box on;
text(0.02, 0.98, 'c)', 'Units', 'normalized', 'FontSize', 12, 'FontWeight', 'bold');

% Subplot 4: Distribuição dos resíduos
subplot(2, 3, 4);

0.8);
xlabel('Resíduo (\chi^2)', 'FontSize', 11, 'FontWeight', 'bold');
ylabel('Frequência', 'FontSize', 11, 'FontWeight', 'bold');
title('Distribuição dos Resíduos', 'FontSize', 12, 'FontWeight', 'bold');
grid on;
hold on;
line([best_residual, best_residual], ylim, 'Color', 'r', 'LineWidth', 2.5, 'LineStyle', '--');
legend({'Distribuição', 'Mínimo'}, 'FontSize', 10, 'Location', 'northwest');
box on;
text(0.02, 0.98, 'd)', 'Units', 'normalized', 'FontSize', 12, 'FontWeight', 'bold');

% Subplot 5: Tabela de Métricas - Simplificada (SEM FUNDO)
subplot(2, 3, 5);
axis off;
hold on;
% Título da tabela
text(0.5, 0.95, 'MÉTRICAS DE AJUSTE', 'FontSize', 14, 'FontWeight', 'bold', ...
 'HorizontalAlignment', 'center', 'Color', [0, 0, 0.5]);
text(0.02, 0.98, 'e)', 'Units', 'normalized', 'FontSize', 12, 'FontWeight', 'bold');
% ========== HELANDER ORIGINAL ==========
text(0.05, 0.85, 'HELANDER ORIGINAL:', 'FontSize', 12, 'FontWeight', 'bold', 'Color', 'r');
metrics_orig = {
 sprintf('R² = %.4f', R2_orig);
 sprintf('χ²_{red} = %.4f', chi2_red_orig);
 sprintf('RMSE = %.0f m/s', RMSE_orig);
 sprintf('Aderência IC = %.1f%%', prop_inside_IC_orig);
};
y_start_orig = 0.78;
for i = 1:length(metrics_orig)
 text(0.1, y_start_orig - (i-1)*0.065, metrics_orig{i}, 'FontSize', 11, 'FontWeight', 'normal');
end
% ========== HELANDER OTIMIZADO ==========
y_start_best = y_start_orig - length(metrics_orig)*0.065 - 0.05;
text(0.05, y_start_best, sprintf('HELANDER OTIMIZADO (iter. %d):', best_iteration), ...
 'FontSize', 12, 'FontWeight', 'bold', 'Color', 'b');
metrics_best = {
 sprintf('R² = %.4f', R2_best);
 sprintf('χ²_{red} = %.4f', chi2_red_best);
 sprintf('RMSE = %.0f m/s', RMSE_best);
 sprintf('Aderência IC = %.1f%%', prop_inside_IC_best);
};
for i = 1:length(metrics_best)
 text(0.1, y_start_best - i*0.065, metrics_best{i}, 'FontSize', 11, 'FontWeight', 'normal');
end
% ========== POLINÔMIO (se disponível) ==========
if exist('R2_best_poly', 'var')
 y_start_poly = y_start_best - (length(metrics_best)+1)*0.065;
 text(0.05, y_start_poly, sprintf('POLINÔMIO (grau %d):', best_degree), ...
 'FontSize', 12, 'FontWeight', 'bold', 'Color', [0, 0.5, 0]);

 metrics_poly = {
 sprintf('R² = %.4f', R2_best_poly);
 sprintf('χ²_{red} = %.4f', chi2_red_best_poly);
 sprintf('RMSE = %.0f m/s', RMSE_best);
 };

 for i = 1:length(metrics_poly)
 text(0.1, y_start_poly - i*0.065, metrics_poly{i}, 'FontSize', 11, 'FontWeight', 'normal');
 end
 y_eval = y_start_poly - (length(metrics_poly)+1)*0.065;
else
 y_eval = y_start_best - (length(metrics_best)+1)*0.065;
end
% ========== AVALIAÇÃO DA ADERÊNCIA ==========
if prop_inside_IC_best > 80
 aderencia_str = '✓ Excelente aderência (>80%)';
 aderencia_color = [0, 0.6, 0];
elseif prop_inside_IC_best > 60
 aderencia_str = '✓ Boa aderência (60-80%)';
 aderencia_color = [0.2, 0.7, 0.2];
elseif prop_inside_IC_best > 40
 aderencia_str = '✓ Moderada aderência (40-60%)';
 aderencia_color = [0.8, 0.6, 0];

aderencia_color);
% Subplot 6: Perfil de velocidade - Comparação com IC do polinômio
subplot(2, 3, 6);

% Área do intervalo de confiança do polinômio (IC 95%) - FAIXA ESCURECIDA
if exist('upper_V_poly', 'var') && exist('lower_V_poly', 'var')

 [0.7, 0.3, 0.3], 'FaceAlpha', 0.7, 'EdgeColor', 'none', ...
